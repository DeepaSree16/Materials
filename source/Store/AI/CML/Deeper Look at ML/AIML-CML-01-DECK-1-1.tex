\begin{frame}[fragile]{Examples of Learned Function or Model}

$ \small x =  \left[\!
     \begin{array}{c}
      x_1 \\
      x_2 \\
       . \\
      x_{d}
      \end{array} 
      \!\right] w =  \left[\!
    \begin{array}{c}
      w_1 \\
      w_2 \\
       . \\
      w_d
      \end{array} 
	\!\right]$ \\[0.8cm]

	\textbf{\large\textbf{ $ f(w,x) = sign(w \cdot x)$}}

\end{frame}
